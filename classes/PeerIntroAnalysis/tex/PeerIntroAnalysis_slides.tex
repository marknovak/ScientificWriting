\section{Format}
\begin{frame}{Analysis of Peer Introductions}

  \textbf{Group format}
  \begin{enumerate}
    \item Work in groups of three.
    \item Read each peer's introduction draft ($\sim$5-10 min each)
    \item Then, for each of your two peers, provide written feedback ($\sim$15 min). (\textit{Questions on next slide.})
    \item After everyone is finished, explain your answers to each peer ($\sim$10 min).
  \end{enumerate}

    \begin{block}{Focus of this activity}
    We are not evaluating scientific correctness.
    We are evaluating clarity, structure, and audience engagement.
  \end{block}

\end{frame}

% ======================================================
\section{Feedback Questions}
\begin{frame}{Feedback Questions}
  \small
  \textbf{For each peer's introduction, answer:}
  \begin{enumerate}
    \item What did the author do well?
    \item Was the topic introduced in a compelling way?  Was the intended audience inferrable and was the topic cast at the right level?  Suggest a means by which it might it be reworked to be more compelling and/or engage a wider audience.
    \item Paraphrase the manuscript's specific question in 1--3 sentences. Was it clear?
    \item Summarize the logicical flow in 4--8 words
      (e.g., x $\rightarrow$ y $\rightarrow$ z). Was the logical flow clear? Identify one place that could be clarified.
    \item If the Introduction ended with a take-home message, what was it? Paraphrase it in 1--3 sentences.  Assess whether it matched your expectations created by the rest of the Introduction.
  \end{enumerate}

\end{frame}

% ======================================================
\section{Debrief}
\begin{frame}{Debrief and Reflection}

  \begin{itemize}
    \item What were common suggestions?
    \item What did you learn about your own writing?
    \item What characterized helpful feedback?
  \end{itemize}

\end{frame}
