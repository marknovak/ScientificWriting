\section{Friendly Reviewers}

\begin{frame}{Pre-Submission Review by Friendly Reviewers}
  \footnotesize
  \medskip
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Selecting Reviewers}
      \begin{itemize}
        \item Choose at least one \textbf{inside}
          the specialty (for technical accuracy) and perhaps 
          one \textbf{outside} it (for clarity and framing)
        \item Avoid/Seek reviewers with strong prior positions on your findings
        \item Prioritize people who will give
          \emph{honest} feedback, not reassurance
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Briefing Reviewers}
      \begin{itemize}
        \item State the target journal and audience
        \item Set a realistic deadline
          (2-3 weeks is standard)
        \item Clarify what kind of feedback is useful:
          structure, argument, clarity, or all three
        \item Identify specific concerns you want addressed
      \end{itemize}
    \end{column}
  \end{columns}

  \medskip
  Thank them by name in the Acknowledgments.

  Consider mentioning in Cover Letter.

  Friendly review is \emph{not} a substitute
  for rigorous self-revision. Send out only
  when you believe the manuscript is essentially complete.

\end{frame}

\section{Final Formatting}
\footnotesize
\begin{frame}
  \frametitle{Final Formatting}

    \textbf{Things I ignore:}
    \begin{itemize}
      \item Font type and size
      \item Margin requirements
      \item Journal-requested citation format
    \end{itemize}

    \textbf{Things I don't ignore:}
    \begin{itemize}
      \item Word/page limits (\textit{incl. title, abstract, main text})
      \item Line spacing (\textit{double}) \& page numbering
      \item Figure placement (\textit{inline except if explicitly specified})
          and formats (\textit{pdf, tiff, eps})
    \item Cover page and/or end-of-manuscript requirements
    \begin{itemize}
      \scriptsize \item Author contribution and Acknowledgments
      \item Conflict of Interest disclosure requirements
      \item Data \& Code Availability statement requirements
    \end{itemize}
    \end{itemize}

    \textbf{Check:}
    \begin{itemize}
    \item Triple-read for spelling and punctuation
    \item Citation completeness and consistency
    \item Journal's double-blind requirements
    \end{itemize}
\end{frame}



\section{Author Contributions}

\begin{frame}{Author Contributions \& the CRediT Taxonomy}

  \textbf{Purpose}
  \smallskip
  \begin{itemize}
    \item Distinguish \textbf{authorship} from \textbf{acknowledgment}
    \item Increase transparency and accountability
    \item Expose \emph{ghost authorship} and \emph{gift authorship}
    \item Increasingly indexed by databases (e.g., OpenAlex)
  \end{itemize}

  \medskip
  \textbf{The CRediT Taxonomy (14 Roles)} \href{https://credit.niso.org}{\texttt{credit.niso.org}}
  \begin{center}
  {\footnotesize
  \begin{tabular}{@{}ll@{}}
    \hline
    Conceptualization     & Methodology \\
    Data Curation         & Project Administration \\
    Formal Analysis       & Resources \\
    Funding Acquisition   & Software \\
    Investigation         & Supervision \\
    Validation            & Visualization \\
    Writing --- Original Draft & Writing --- Review \& Editing \\
    \hline
  \end{tabular}
  }
\end{center}

\end{frame}

\begin{frame}{Author Contributions Statement}

      \textbf{Common Formats}
      \medskip
      \begin{enumerate}
        \item \textbf{Prose (narrative)}\\
          \small ``J.S. conceived the study and led data collection.
          M.R. performed statistical analysis\ldots''

        \bigskip

        \item \textbf{Role-tagged list}\\
          \small \textit{Conceptualization}: J.S., M.R.\\
          \textit{Formal Analysis}: M.R.\\
          \textit{Writing--Original Draft}: J.S.\\
          \textit{Writing--Review \& Editing}: All authors
      \end{enumerate}

      \bigskip

      \alert{\textbf{Imperative:}} Agree on contributions \emph{before drafting}, not retroactively. 
      Disagreements are common, uncomfortable, and potentially costly.%
      \footnote{See \emph{Collaboration} class.}

\end{frame}

\section{Open Science}

\begin{frame}{Open Science}
  \begin{itemize}
    \item Promotes \textbf{transparency} and \textbf{reproducibility}
    \item Increases citation impact and research visibility
    \item Broadens engagement across the research community
    \item Enhances credibility
  \end{itemize}

  \medskip
  \textbf{Key Considerations}
  \begin{itemize}
    \item Article Processing Charges (APCs) can be substantial
    \item Data sensitivity and participant privacy constraints
    \item Institutional and funder mandates increasingly require OA
  \end{itemize}

\end{frame}


\begin{frame}{Open Access Paper Publication Models}

  \smallskip
  {\footnotesize
  \begin{tabular}{@{}lp{2.2cm}p{5.8cm}@{}}
    \hline
    \textbf{Type} & \textbf{Who Pays} & \textbf{Description} \\
    \hline
    \textbf{Gold}    & Author / funder   & Published OA immediately by the journal; APC required \\
    \textbf{Green}   & Nobody            & Author self-archives a preprint or accepted manuscript in a repository (e.g., bior$\chi$iv) \\
    \textbf{Diamond} & Nobody            & Community- or institution-funded; no fees for authors or readers \\
    \textbf{Hybrid}  & Author            & Subscription journal offers OA for individual articles upon APC payment \\
    \textbf{Bronze}  & Nobody            & Freely readable on the publisher site \emph{without} open licence; access can be revoked \\
    \hline
  \end{tabular}
  }

  \medskip
  \small
  \textbf{Note:} Green OA costs nothing and is compatible with most journals.\\
  Check policies at \href{https://openpolicyfinder.jisc.ac.uk}{\texttt{https://openpolicyfinder.jisc.ac.uk}}.

\end{frame}



\begin{frame}
  \frametitle{Data \& Code Sharing}
  \begin{itemize}
    \item Increasingly required by journals and funders
    \item Can be cited - Digitial Object Identifier  (DOI)
    \item Common repositories:
    \begin{itemize}
      \item Data: Zenodo, FigShare, Data One
      \item Code: GitHub, Zenodo
      \item Supplementary materials (\textit{no longer considered enough})
    \end{itemize}
    \item Best practices:
    \begin{itemize}
      \item Include comprehensive documentation and README files
      \item Provide appropriate licensing
    \end{itemize}
  \end{itemize}
\end{frame}


\begin{frame}{FAIR Data Principles}

  \textbf{FAIR} defines standards to maximise reuse and reproducibility.
  \medskip
  {\footnotesize
  \begin{description}
    \item[\textbf{F --- Findable}] Data are assigned persistent identifiers (e.g., DOIs)
      and described with rich, searchable metadata.
    \item[\textbf{A --- Accessible}] Data and metadata are retrievable via open, standardised
      protocols; access conditions are clearly stated.
    \item[\textbf{I --- Interoperable}] Data use common formats, vocabularies, and ontologies
      to enable integration with other datasets and tools.
    \item[\textbf{R --- Reusable}] Data carry explicit licences (e.g., CC BY) and sufficient
      provenance information for others to build on them.
  \end{description}
  }

  \medskip
  \small
  Full guidelines at \href{https://www.go-fair.org/fair-principles}{\texttt{go-fair.org/fair-principles}}.\\
  \smallskip
  \textbf{Note:} FAIR does not require data to be \emph{open} --- sensitive data can be
  FAIR with controlled access.

\end{frame}

\section{Preprint Repositories}

\begin{frame}{Preprint Repositories: Examples}

  \footnotesize
  \begin{description}
    \item[\href{https://www.biorxiv.org}{\textbf{bioR$\chi$iv}}]
      Life sciences; operated by Cold Spring Harbor Laboratory;
      the dominant biology preprint server since 2013.

    \item[\href{https://ecoevorxiv.org}{\textbf{EcoEvoR$\chi$iv}}]
      Ecology and evolutionary biology; community-governed;
      supports registered reports and post-publication versions.

    \item[\href{https://arxiv.org}{\textbf{ar$\chi$iv}}]
      Physics, mathematics, computer science, quantitative biology;
      the oldest major preprint server (est.\ 1991); moderated by
      field endorsers.

    \item[\href{https://zenodo.org}{\textbf{Zenodo}}]
      General-purpose; accepts preprints, datasets,
      code, and presentations; assigns DOIs; widely used for
      open science compliance.

    \item[\href{https://www.authorea.com}{\textbf{Authorea}}]
      Collaborative writing and preprint platform; supports
      live figures, data embedding, and inline commenting;
      blurs the boundary between manuscript and publication.
  \end{description}

\end{frame}

\begin{frame}{Preprint Repositories: Primary Pros \& Cons}

  \footnotesize
  \begin{columns}[T]

    \begin{column}{0.48\textwidth}
      \textbf{Advantages}
      \smallskip
      \begin{itemize}
        \item \textbf{Speed:} findings are public immediately,
          months before formal publication
        \item \textbf{Priority:} timestamped record of contribution
        \item \textbf{Visibility:} indexed by Google Scholar, etc.
        \item \textbf{Feedback:} community comments before peer review
        \item \textbf{Open access:} freely available
        \item \textbf{Funder compliance:} satisfies some open
          science mandates immediately
      \end{itemize}
    \end{column}

    \begin{column}{0.48\textwidth}
      \textbf{Disadvantages}
      \smallskip
      \begin{itemize}
        \item \textbf{Journal policies:} some journals
          discourage or prohibit
        \item \textbf{No peer review:} errors and weak claims
          are public
        \item \textbf{Perception:} some fields and hiring
          committees do not credit preprints
        \item \textbf{Retraction gap:} flawed preprints may
          persist even after journal rejection
        \item \textbf{Media risk:} premature press coverage
          before validation
      \end{itemize}
    \end{column}

  \end{columns}

\end{frame}


\begin{frame}{Peer Community In$\ldots$ (PCI)}

\begin{columns}[T, totalwidth=\textwidth]

\column{0.8\textwidth}

  \footnotesize
  \href{https://peercommunityin.org}{\texttt{PeerCommunityIn.org}}

  \medskip
  \textbf{Goals}
  \begin{itemize}
    \item Transparent peer review \emph{before}
      (or \emph{instead of}) journal submission
    \item Eliminate fees for authors and readers
    \item Reduce dependence on commercial publishers
  \end{itemize}

  \medskip
  \textbf{How It Works}
  \begin{itemize}
    \item Authors submit preprint (e.g., to bioR$\chi$iv), then
      request PCI review
    \item Volunteer recommender (acting editor) solicits and coordinates reviews
    \item Accepted preprints receive a \textbf{PCI Recommendation} ---
      citable endorsement
    \item PCI-friendly journals accept
      recommended preprints, or regular review
  \end{itemize}

\section{Page Charges}

\begin{frame}{Page Charges \& Publication Fees}

  \footnotesize
  \begin{itemize}
    \item \textbf{Article Processing Charges (APCs):} \$500--\$12,000+
      depending on journal/OA model
    \item \textbf{Page charges:}  \$0--\$200 per printed page for
      subscription journals
    \item \textbf{Color figure fees:} \$300--\$1,000+ per figure;
      often free online-only
  \end{itemize}

  \medskip
  \textbf{Who Pays --- and How to Find Out}
  \begin{itemize}
    \item \textbf{Ask your advisor first:} most PIs budget for publication
      costs through grants; this is a normal conversation to have
    \item \textbf{Institutional waivers:} many universities (but not OSU!) have
      OA publishing funds; check with library
    \item \textbf{Funder mandates:} NIH, Wellcome Trust, and others
      \emph{require} OA and may cover fees directly
    \item \textbf{Request a waiver:} most journals offer hardship waivers
      for authors without funding --- ask before assuming you must pay
  \end{itemize}

  \medskip
  \alert{\textbf{Bottom line:}} You shouldn't be expected to pay out of pocket.
  Clarify funding \emph{before} submission, not after acceptance.

\end{frame}
