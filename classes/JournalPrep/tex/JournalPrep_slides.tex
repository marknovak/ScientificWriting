\section{Final Formatting}

\begin{frame}
  \frametitle{Final Formatting}
  \begin{itemize}
    \item Verify all formatting matches journal specifications:
    \begin{itemize}
      \item Font type and size
      \item Line spacing (single, 1.5, or double)
      \item Margin requirements
      \item Page numbering
    \end{itemize}
    \item Check figure and table formatting:
    \begin{itemize}
      \item Resolution (minimum 300 dpi for publication)
      \item Captions and labels
      \item File formats (PDF, TIFF, EPS)
    \end{itemize}
    \item Review references for consistency
    \item Proofread for spelling and punctuation
  \end{itemize}
\end{frame}

\section{Journal-Specific Author Guidelines}

\begin{frame}
  \frametitle{Journal-Specific Author Guidelines}
  \begin{itemize}
    \item Consult the journal's ``Instructions for Authors''
    \item Key elements to check:
    \begin{itemize}
      \item Article types and word/page limits
      \item Abstract length and structure
      \item Keywords requirements
      \item Section organization expectations
      \item Citation format
    \end{itemize}
    \item Check author contribution and funding acknowledgments
    \item Verify conflict of interest disclosure requirements
    \item Review data availability statement requirements
  \end{itemize}
\end{frame}

\section{Open Science}

\begin{frame}
  \frametitle{Open Science}
  \begin{itemize}
    \item Open science promotes transparency and reproducibility:
    \begin{itemize}
      \item Open access publishing
      \item Preprints and preregistration
      \item Open peer review
      \item Open evaluation
    \end{itemize}
    \item Benefits:
    \begin{itemize}
      \item Increased citation impact
      \item Broader research community engagement
      \item Enhanced credibility
    \end{itemize}
    \item Considerations:
    \begin{itemize}
      \item Publication fees for open access
      \item Data sensitivity and privacy concerns
    \end{itemize}
  \end{itemize}
\end{frame}

\section{Data \& Code Sharing}

\begin{frame}
  \frametitle{Data \& Code Sharing}
  \begin{itemize}
    \item Increasingly required by journals and funders
    \item Repositories for sharing:
    \begin{itemize}
      \item Data: Zenodo, OSF, discipline-specific repositories
      \item Code: GitHub, GitLab, Zenodo
      \item Supplementary materials: institutional repositories
    \end{itemize}
    \item Best practices:
    \begin{itemize}
      \item Use persistent identifiers (DOI)
      \item Include comprehensive documentation and README files
      \item Provide appropriate licensing
      \item Ensure anonymization where needed
    \end{itemize}
    \item Data availability statements must specify locations
  \end{itemize}
\end{frame}

\section{Submission Logistics}

\begin{frame}
  \frametitle{Submission Logistics}
  \begin{itemize}
    \item Prepare submission materials:
    \begin{itemize}
      \item Main manuscript document
      \item Figures and tables
      \item Supplementary materials
      \item Cover letter
    \end{itemize}
    \item Cover letter content:
    \begin{itemize}
      \item Significance of the work
      \item Confirmation that work is original and not under review elsewhere
      \item Suggested reviewers (if permitted)
    \end{itemize}
    \item Use journal's online submission system
    \item Track manuscript number and submission confirmation
    \item Plan for editorial timelines and revision rounds
  \end{itemize}
\end{frame}

\section{Page Charges}

\begin{frame}
  \frametitle{Page Charges \& Publication Fees}
  \begin{itemize}
    \item Traditional journals may assess page charges:
    \begin{itemize}
      \item Usually charged per page
      \item Costs vary widely by journal
      \item May include color figure charges
    \end{itemize}
    \item Open access journals typically charge article processing fees (APCs)
    \begin{itemize}
      \item Range: \$0 to \$5,000+
      \item Often discounted for specific regions or authors
    \end{itemize}
    \item Check funding sources:
    \begin{itemize}
      \item Many grants include publication budgets
      \item Institutional memberships may provide discounts
    \end{itemize}
    \item Compare costs across potential target journals
  \end{itemize}
\end{frame}
