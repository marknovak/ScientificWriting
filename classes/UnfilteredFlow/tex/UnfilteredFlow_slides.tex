\begin{frame}{Unfiltered Flow: High-Velocity Drafting}



\textbf{Goals}
\begin{itemize}
    \item Reduce perfectionism and lower barriers to starting
    \item Identify core ideas by writing without premature editing
\end{itemize}

\textbf{Approach}
\begin{itemize}
    \item Short, timed writing sprints
    \item No deleting, no revising
    \item Focus on momentum
\end{itemize}

\section{Warm-Up}

\subsection{Zero Filter}

\begin{frame}{Zero Filter}
\begin{block}{Prompt}
What is the worst possible first sentence you could write for a scientific paper?
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 5 minutes
    \item Do not aim for quality; lean into absurdity or imperfection.  Write badly!

\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Reduce perfectionism and cognitive inhibition
    \item Prime fluency and lower the barrier to starting
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item What did it feel like to deliberately write badly?
    \item Did removing quality expectations change your writing speed or tone?
\end{itemize}
\end{frame}

\subsection{Stream of Consciousness}

\begin{frame}{Stream of Consciousness}
\begin{block}{Prompt}
Write everything you already know about your research topic---no structure.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 5 minutes
    \item No deleting or revising
    \item If stuck, repeat the last word until a new idea emerges
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Externalize tacit knowledge
    \item Reveal gaps or redundancies in understanding
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item What surprised you about what you do (or do not) know?
    \item Did any new connections emerge?
\end{itemize}
\end{frame}

\section{Clarity and Velocity}

\subsection{Jargon Free}

\begin{frame}{Jargon Free}
\begin{block}{Prompt}
Explain your research question without using any technical jargon.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 8 minutes
    \item Avoid field-specific terminology entirely
    \item If you accidentally use jargon, immediately rephrase and continue
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Improve conceptual clarity and accessibility
    \item Strengthen introductions and broader-impact framing
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Which concepts were hardest to simplify?
    \item How might this improve your manuscript's readability?
\end{itemize}
\end{frame}

\subsection{Volume Competition}

\begin{frame}{Volume Competition}
\begin{block}{Prompt}
Write as much as possible describing your study system or methods.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 8 minutes
    \item Word count determines the winner
    \item Content must remain at least minimally coherent
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Maintain baseline coherence under pressure
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Did speed compromise clarity?
    \item What strategies helped you maintain momentum?
\end{itemize}
\end{frame}

\section{Framing and Structure}

\subsection{Reverse Logic}

\begin{frame}{Reverse Logic}
\begin{block}{Prompt}
Argue why your central hypothesis is wrong.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 10 minutes
    \item Construct the strongest possible counterargument
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Sharpen critical thinking
    \item Anticipate reviewer critiques
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Did this reveal any real weaknesses?
    \item How could you preempt these critiques in your manuscript?
\end{itemize}
\end{frame}

\subsection{Methods in Motion}

\begin{frame}{Methods in Motion}
\begin{block}{Prompt}
Describe your methods section as if explaining it to a colleague who wants to replicate your work.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 10 minutes
    \item Focus on sequence and logic, not polish
    \item Keep typing even if details feel incomplete
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Translate procedural knowledge into structured prose
    \item Identify missing steps or ambiguities
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Where did you hesitate or lack detail?
    \item What parts felt most clear vs. underdeveloped?
\end{itemize}
\end{frame}

\subsection{Figure Narrative}

\begin{frame}{Figure Narrative}
\begin{block}{Prompt}
Write the results section for one key figure in your manuscript.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 12 minutes
    \item Describe patterns, trends, and interpretations
    \item Do not reference the figure explicitly; make the text stand alone
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Strengthen the link between data and narrative interpretation
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Did the writing clearly convey the main result?
    \item What additional analysis or visualization might strengthen this section?
\end{itemize}
\end{frame}

\subsection{Abstract Compression}

\begin{frame}{Abstract Compression}
\begin{block}{Prompt}
Draft a complete abstract for your manuscript.
\end{block}

\vspace{0.8em}
\textbf{Rules}
\begin{itemize}
    \item 12 minutes
    \item Must include context, question, methods, key results, and implications
    \item Prioritize completeness over perfection
\end{itemize}

\textbf{Intention}
\begin{itemize}
    \item Synthesize a coherent high-level narrative
\end{itemize}

\textbf{Debrief}
\begin{itemize}
    \item Which component was hardest to articulate quickly?
    \item How does this draft compare to your current abstract (if one exists)?
\end{itemize}
\end{frame}
