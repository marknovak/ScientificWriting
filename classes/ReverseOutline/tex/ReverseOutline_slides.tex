\begin{frame}{What Is a Reverse Outline?}
  Instead of planning what you want to say, map what you actually said.

  \medskip
  \begin{block}{Diagnose structural problems}
    Redundancy, misplacement, missing links, and logical gaps.
  \end{block}

  \medskip
  \textbf{Reverse outlining reveals paragraphs that:}
  \begin{itemize}
    \item drift from the intended argument
    \item mix multiple ideas that should be separated
    \item do not clearly support the central claim
  \end{itemize}

 \end{frame}

% ============================================================
\begin{frame}{The Process}

  \begin{enumerate}
    \item Analyze your manuscript paragraph by paragraph.
    \item Write a short summary of each paragraph's \textbf{purpose}.
    \item Use summaries as a \textbf{structural map} of the manuscript.
  \end{enumerate}
  
  \medskip

  For each paragraph, answer two core questions:

  \medskip
  \begin{block}{1) What it says (\emph{Topic})}
    The main idea or point being made.
  \end{block}

  \begin{block}{2) What it does (\emph{Function})}
    How the paragraph serves the manuscript, its argument, and your audience.
  \end{block}

\end{frame}

% ============================================================
\begin{frame}{Activity: Reverse Outline Your Paragraphs}

  \footnotesize
  \textbf{Part 1} (up to 20 minutes)
  \begin{itemize}
    \item  Reverse outline each paragraph as objectively as possible.
    \item Do not edit yet, just take notes.
  \end{itemize}

  \medskip
  \textbf{Part 2} (up to 10 minutes)
  \begin{itemize}
    \item  Share your paragraphs with your neighbour.
    \item Reverse outline each others' paragraphs.
  \end{itemize}

  \medskip
  \textbf{Part 3} (up to 15 minutes)
  \begin{itemize}
    \item Use both reverse outlines to assess whether each paragraph's sentences say what they need to say and do.
  \end{itemize}

  \medskip
  \textbf{Part 4} (up to 15 minutes)
  \begin{itemize}
    \item Shift to a manuscript-level perspective.
    \item Assess how each paragraph, as a whole, serves the manuscript
      and its audience.
  \end{itemize}

  \medskip
  We'll use the remaining class period to reflect as a group or continue reverse outlining additional paragraphs.

\end{frame}
