\begin{frame}{What Is a Reverse Outline?}
  Instead of planning what you want to say, you map what you
  actually said.

  \medskip
  \begin{block}{Why this matters}
    Diagnose hard to see structural problems:
    redundancy, misplacement,  missing links, and logical gaps.
  \end{block}

  \medskip
  \textbf{What reverse outlining reveals:}
  \begin{itemize}
    \item Paragraphs that drift from the intended argument.
    \item Paragraphs that mix multiple ideas.
    \item Paragraphs that do not clearly support the central claim.
  \end{itemize}

  \medskip
  \textbf{The process:}
  \begin{itemize}
    \item Read a draft paragraph by paragraph.
    \item Write a short summary of each paragraph's purpose.
    \item Use summaries as a structural map of the manuscript.
  \end{itemize}



\end{frame}

% ============================================================
\begin{frame}{Two Core Questions Per Paragraph}

  For each paragraph, identify:

  \medskip
  \begin{block}{1) What it says (\emph{Topic})}
    The main idea or point being made.
  \end{block}

  \begin{block}{2) What it does (\emph{Function})}
    How the paragraph serves the manuscript and its argument.
  \end{block}

\end{frame}

% ============================================================
\begin{frame}{Activity}

  \footnotesize
  \textbf{Part 1}
  \begin{itemize}
    \item Produce a reverse outline for each paragraph as objectively
      as possible.
    \item Do not edit yet; keep notes if needed.
    \item Time: up to 30 minutes.
  \end{itemize}

  \medskip
  \textbf{Part 2}
  \begin{itemize}
    \item Use your reverse outline to assess whether paragraph sentences
      say what they need to say and do for the manuscript.
    \item Time: up to 15 minutes.
  \end{itemize}

  \medskip
  \textbf{Part 3}
  \begin{itemize}
    \item Shift to a manuscript-level perspective.
    \item Assess how each paragraph, as a whole, serves the manuscript
      and its audience.
    \item Time: up to 15 minutes.
  \end{itemize}

  \medskip
  We'll use the remaining class period to reflect as a group and
  reverse outline additional paragraphs from your manuscripts.

\end{frame}
