\section{Audience \& Journal}

\begin{frame}
  \frametitle{Audience Identification}
  \begin{itemize}
    \item Know your target audience:
    \begin{itemize}
      \item General scientific audience
      \item Specialists vs.\ interdisciplinary researchers
      \item Theoreticians and/or empiricists
    \end{itemize}
    \item Audience shapes decisions about:
    \begin{itemize}
      \item Terminology and background explanation
      \item Context and significance framing
      \item Level of technical detail
      \item Visual presentation choices (e.g., conceptual figs)
    \end{itemize}
    \item Tailor writing to match audience expertise and interests
    \item Consider what audience knows vs.\ needs to know
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Journal Selection}
  \begin{itemize}
    \item Identify target journals \emph{before} you start writing
    \item Rank potential journals:
    \begin{itemize}
      \item Aspirational journals (high impact, broad audience)
      \item Realistic journals (good fit for topic and scope)
      \item Backup journals (alternative options if rejected)
    \end{itemize}
    \item Consider:
    \begin{itemize}
      \item Similarity of published articles to your work
      \item Journal scope and audience
      \item {\tiny Open access policies (and costs)}
      \item {\tiny Impact factor and prestige (with caution)}
    \end{itemize}
    \item If you're not confident
    \begin{itemize}
      \item Check target journal requirements for alignment
      \item Review recent issues
    \end{itemize}
    \end{itemize}
\end{frame}

\section{Organization and Flow}

\begin{frame}
  \frametitle{Signposting}
  \begin{itemize}
    \item Guide readers through your argument with effective signposting
    \item Use transitions and connectives:
    \begin{itemize}
      \item Topic sentences that signal direction
      \item Transitional phrases linking paragraphs and sections
      \item Explicit statements of relationships between ideas
    \end{itemize}
    \item Employ structural markers:
    \begin{itemize}
      \item Clear section headings and subheadings
      \item Summary sentences to recap key points
    \end{itemize}
    \item Help readers anticipate and understand:
    \begin{itemize}
      \item What comes next
      \item How ideas connect
      \item Why information matters
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Structural Coherence}
  \begin{itemize}
    \item Maintain logical flow throughout the paper
    \item Structure provides the ``roadmap'' for readers
    \item Key aspects:
    \begin{itemize}
      \item Logical progression of ideas within sections
      \item Clear connections between sections
      \item Motivation established before presenting methods
      \item Results presented before interpretation
      \item Conclusions grounded in findings
    \end{itemize}
    \item Typical structure (IMRaD):
    \begin{itemize}
      \item Introduction: context and questions
      \item Methods: how you answered questions
      \item Results: what you found
      \item Discussion: what it means
    \end{itemize}
    \item Ensure each section connects to and builds on previous
  \end{itemize}
\end{frame}



\section{Content Strategies}

\begin{frame}
  \frametitle{Methods, Results \& Supplementary Materials}
  \textbf{Methods and Results}
  \begin{itemize}
    \item Sufficient detail for general understanding \& reproducibility
    \item Balance completeness with readability
    \item Use parallel subheadings to organize
  \end{itemize}

  \textbf{Supplementary Materials}
  \begin{itemize}
    \item Use strategically to keep main text focused and concise
    \item Main text must remain complete and independent
    \item Include:
    \begin{itemize}
      \item Extended methods, secondary or alternative analyses
      \item Statistical analysis details or results tables
      \item Details and results that are not essential to main narrative
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Integration of Equations}
  \begin{itemize}
    \item Introduce equations before presenting them
    \item Interpret equations after presenting them
    \item Equations (symbols) are words in a sentence, not standalone objects
    {\footnotesize(Use appropriate grammar and punctuation)}
    \item Guidelines:
    \begin{itemize}
      \item Define all variables and parameters
      \item Explain the logic and significance of equations to broad audience
      \item Follow standard (or explain) mathematical notation
    \end{itemize}
    \item Keep equations in the main text only if essential to narrative%
    \footnote{See Fawcett \& Higginson, \emph{PNAS} 2012 and responding commentaries}
  \end{itemize}
\end{frame}

\begin{frame}{Designing an Effective Title}

  Most-read part --- determines if reader even looks at paper!

  \footnotesize
  \medskip
  \begin{itemize}
    \item \textbf{Be specific:} state the finding, not the topic\\

    \item \textbf{Aim for 8--14 words}: enough to be precise,
      short enough to scan
    \item \textbf{Front-load key terms} for visibility\\
      {\scriptsize Avoid: 
      \textit{``A study of\ldots''}, 
      \textit{``The role of\ldots''}, 
      \textit{``Effects of X on Y''}
      }
    \item \textbf{Consider style:} declarative sentence,
      question, or noun phrase\\
      {\scriptsize A noun phrase has no main verb (essentially a sophisticated label).}
      {\scriptsize \emph{"Temperature dependence of..."} names phenomenon without asserting anything about it.}
    \item \textbf{Catchy, clear, concise}%
    \footnote{See \href{../../readings/pdfs/CatchyClearConcise.pdf}{Singh Chawla (2025)}}
  \end{itemize}

        {\scriptsize
      \begin{tabular}{@{}lp{8cm}@{}}
        \hline
        \textit{Weak}   & \textit{A Study of predator foraging behavior
                          in an aquatic system} \\
        \textit{Stronger} & \textit{Temperature mediates Type II functional
                          responses in a dragonfly--Daphnia predator--prey
                          system} \\
        \textit{Strongest} & \textit{Temperature dependence of attack rates but not
                          handling times in a dragonfly--Daphnia
                          predator--prey system} \\
        \hline
      \end{tabular}
      }

\end{frame}


\begin{frame}{Structuring an Effective Abstract}

  \footnotesize
  \begin{description}
    \setlength{\itemsep}{3pt}
    \item[\textbf{1--2 sentences}] Broad introduction to the field,
      accessible to any scientist
    \item[\textbf{2--3 sentences}] Detailed background for scientists
      in related disciplines
    \item[\textbf{1 sentence}] The specific problem this study addresses
    \item[\textbf{1 sentence}] The main result --- use \emph{``here we
      show''} or equivalent
    \item[\textbf{2--3 sentences}] What the result reveals relative to
      prior understanding; what it adds to existing knowledge
    \item[\textbf{1--2 sentences}] Broader context and implications
  \end{description}

  \medskip
  \textbf{Key Principles}%
  \footnote{See \textit{Nature-Summary-Paragraph.pdf}}
  \begin{itemize}
    \item Self-contained: interpretable without access to the full paper
    \item Concise: limit to essential details and key message
    \item Include quantitative results where possible
      {\scriptsize (e.g., \textit{``attack rates declined by 43\% at 10$^\circ$C
      relative to 20$^\circ$C, with handling time showing no change.''})}
    \item State implications explicitly: do not leave the reader
      to infer them
  \end{itemize}

\end{frame}

\section{Additional Sand-Jensen Principles}

\begin{frame}{Write Concisely}

  \footnotesize
  Brevity signals confidence in argument --- length can signal the opposite

  \medskip
  \textbf{Principles}
  \begin{itemize}
    \item \textbf{One idea per sentence;} one purpose per paragraph
    \item \textbf{Cut throat-clearing:} 1st sentence can often be deleted
    \item \textbf{Prefer the shorter word:} \textit{``use"} not
      \textit{``utilize"}; \textit{``show"} not \textit{``demonstrate"}
    \item \textbf{Avoid redundancy:} \textit{``completely eliminate''},
      \textit{``end result''}, \textit{``past history''}
  \end{itemize}

  \medskip
  \textbf{A Useful Test}\\
  \smallskip
  After drafting, ask: \emph{does every sentence earn its place?}
  If a sentence neither advances the argument nor provides essential
  context, delete it.

  \medskip
  {\scriptsize
  \begin{tabular}{@{}lp{8cm}@{}}
    \hline
    \textit{Verbose}  & \textit{``It is important to note that the results
                        of the analysis that was conducted clearly
                        demonstrate that\ldots''} \\
    \textit{Concise}  & \textit{``The analysis shows that\ldots''} \\
    \hline
  \end{tabular}
  }

\end{frame}


\begin{frame}{Discussion Writing: Implications and Speculation}

  \footnotesize
  \textbf{The Discussion is Where Science Moves Forward}\\
  \smallskip
  Stating only what you found is not enough. Readers need to know
  what your findings \emph{mean} and where they lead.

  \medskip
  \textbf{A strong discussion $\ldots$}
  \begin{itemize}
    \item \textbf{Answers the question posed in the Introduction }---
      return explicitly to hypothesis or objective
    \item \textbf{Compares findings to prior work} --- not just
      \textit{``consistent with''} but \textit{why} agreement
      or disagreement matters
    \item \textbf{States implications directly} --- do not leave
      the reader to infer
    \item \textbf{Licenses informed speculation} --- clearly flagged
      (\textit{``we suggest\ldots''}, \textit{``one possibility
      is\ldots''}) but not avoided
    \item \textbf{Identifies limitations honestly} without
      undermining the contribution
    \item \textbf{Opens new questions} --- the best papers leave
      a field more curious than they found it
  \end{itemize}

  \medskip
  Watson \& Crick (1953): \textit{``It has not escaped
  our notice that the specific pairing we have postulated immediately
  suggests a possible copying mechanism for the genetic material.''}
  Had they omitted this implication, acceptance of the model
  would have been substantially delayed.

\end{frame}


\begin{frame}{Figure-Driven Storytelling}

  \footnotesize
  \textbf{Figures Are Not Decoration}\\
  \smallskip
  A well-designed figure conveys structure, mechanism, or pattern
  that would require thousands of words to describe --- and is
  more memorable.

  \medskip
  \begin{itemize}
    \item \textbf{Plan figures before writing:} if you can sketch
      your story as a sequence of figures, your narrative structure
      is sound
    \item \textbf{Each figure should make one clear claim} ---
      if it requires extensive caption explanation, simplify
    \item \textbf{Figures and text must be mutually reinforcing,}
      not redundant --- do not describe in prose what is already
      visible in the figure
    \item \textbf{Captions should be self-contained:} a reader
      should understand the figure without reading the main text
    \item \textbf{Order figures to mirror narrative arc} ---
      the figure sequence is a parallel story
  \end{itemize}

  \medskip
  \textbf{Use figures creatively and purposefully}\\
  \smallskip
  {\scriptsize
  \begin{tabular}{@{}ll@{}}
    \hline
    \textbf{Conceptual diagrams}  & Hypotheses, frameworks, food webs \\
    \textbf{Data figures}         & Results; patterns across treatments or gradients \\
    \textbf{System figures}       & Study site, organism, or experimental design \\
    \textbf{Synthesis figures}    & Meta-analytic summaries; effect size plots \\
    \hline
  \end{tabular}
  }

\end{frame}


\begin{frame}{Managing Jargon and Terminology}

  \footnotesize
  \textbf{Jargon is a double-edged sword} \\
  Exists to enable precision, not to signal membership. 
  Overuse narrows audience, obscures weak reasoning,
  and slows reader.

  \medskip
  \begin{itemize}
    \item \textbf{Define terms on first use} --- even for a
      specialist audience
    \item \textbf{Use abbreviations sparingly} --- only
      for terms used $\gtrsim 5$ times
    \item \textbf{Calibrate to audience} --- what is jargon
      for general reader may be essential shorthand for
      specialist
    \item \textbf{Be especially careful in:} the Abstract (broadest
      readership), Introduction (context-setting), and Discussion
      (implications must be accessible)
  \end{itemize}

  \medskip
  {\scriptsize
  \begin{tabular}{@{}lp{7.5cm}@{}}
    \hline
    \textit{Jargon-heavy} & \textit{``Spatiotemporal heterogeneity
      in resource availability mediates trophic cascade attenuation
      across productivity gradients.''} \\
    \textit{Clearer} & \textit{``Patchy resource availability weakens
      top-down control by predators in more productive systems.''} \\
    \hline
  \end{tabular}
  }

\end{frame}

\section{Paper Analysis}

\begin{frame}
  \frametitle{Analysis of Effective Papers}
  \begin{itemize}
    \item Identify patterns in engaging papers
    \item Note what works and why
    \item Analyze overall structure and organization:
    \begin{itemize}
      \item How do authors present their problem?
      \item What background information do they provide?
      \item How do they organize results?
      \item How do they connect findings to broader context?
    \end{itemize}
    \item Examine writing:
    \begin{itemize}
      \item Sentence construction and clarity
      \item Use of active vs.\ passive voice
      \item Paragraph organization and transitions
      \item Balance of technical detail and accessibility
    \end{itemize}
    \item Model your own writing on effective examples
  \end{itemize}
\end{frame}
