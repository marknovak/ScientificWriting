\section{Function}
\begin{frame}{What Cover Letters Do}

  \footnotesize
  \textbf{Strategic Functions:}
  \begin{itemize}
    \item \textbf{Provide context:} Summarize the key contribution and significance
    \item \textbf{Journal fit:} Convince the editor your work belongs in their journal
    \item \textbf{Frame the manuscript:} Guide the editor's interpretation before peer review
    \item \textbf{Manage the review:} Disclose conflicts, acknowledge related work
    \item \textbf{Professional communication:} First impression with an editor
  \end{itemize}

  \medskip
  \begin{block}{Key Principle}
    Even though the manuscript contains the scientific argument, the cover letter
    \textbf{frames it} for the editor---it's often the first thing they read.
  \end{block}

\end{frame}

\section{Structure}
\begin{frame}{Suggested Structure}


  Each paragraph has a specific strategic function.

  \medskip

  \begin{enumerate}
    \item \textbf{Opening:} Introduce manuscript + central finding + broader significance
    \item \textbf{Contribution:} How the research supports your main claim
    \item \textbf{Journal Fit:} Why this manuscript belongs in this journal
    \item \textbf{Compliance:} Required editorial declarations
    \item \textbf{Closing:} Thank the editor + signature
  \end{enumerate}

  \medskip

  Try keep it under 1 page. Be concise and professional.

\end{frame}

\section{Activity}

\begin{frame}{Today's Activity}

  Groups of 3-4 people.

  \medskip

  \begin{enumerate}
    \item Share your reverse outlines: \textit{what structure(s) did you observe?}
    \item Compare to my template: \textit{similarities and differences?}
    \item Discuss strategic choices: \textit{Why organize information these ways?}
    \item \textit{Outline} your own cover letter. 
    \begin{enumerate}
      \item High-level: What structure will you use? What information will go in each paragraph?
      \item Low-level: What specific points will you make in each paragraph?
    \end{enumerate} 
    \item Start \textit{drafting} your letter (time permitting).
  \end{enumerate}

  \medskip
  \begin{block}{Remember}
    The template and examples are only a starting point. Personalize them to match your voice.
  \end{block}

\end{frame}
