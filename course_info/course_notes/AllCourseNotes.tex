% This script will pull the tex-created pdfs of all specified classe notes together into one pdf container

\documentclass{beamer}
\usepackage[utf8]{inputenc}
\usepackage{pdfpages}
\usepackage{bookmark}

\setbeamertemplate{section in toc}[sections numbered]
\setbeamertemplate{subsection in toc}[subsections numbered]


\newcommand{\addsectionpdf}[1]{%
    \section{#1}
    \includepdf[pages=-]{../../classes/#1/tex/#1_slides.pdf}
    \begin{frame}
        \tableofcontents
    \end{frame}
}

\title{\textbf{IB 514: Scientific Writing}\\{\footnotesize \textit{All course slides strung together}}}

\begin{document}
    \begin{frame}
        \maketitle
    \end{frame}

    \begin{frame}
        \tableofcontents
    \end{frame}

    \addsectionpdf{Introduction}
    \addsectionpdf{Habits}
    \addsectionpdf{Workflows}
    \addsectionpdf{MetaStructure}
    \addsectionpdf{Literature}
    \addsectionpdf{Collaboration}
    \addsectionpdf{PeerReview}
    \addsectionpdf{JournalPrep}

\end{document}
